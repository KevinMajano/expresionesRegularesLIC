Listado de Expresiónes Regulares

//Expresión regular del correo

/^[-\w.%+]{1,64}@(?:[A-Z0-9-]{1,63}\.){1,125}[A-Z]{2,63}$/i

//Expresión regular del DUI

/^\d{8}[-][0-9]$/

//Expresión regular del Telefono

/^\d{4}[-]\d{4}$/

//Expresión regular de una matricula

/^[A-Z]{1,2}\s\d{4}\s([B-D]|[F-H]|[J-N]|[P-T]|[V-Z]){2,3}$/

//Expresión regular de numeros

/^-?(?:\d+(?:.\d*)?)$/

//Expresión regular de URL

/^(ftp|http|https):\/\/[^ "]+$/

//Expresión regular del Carnet

/^['A-Z']{2}\d{8}$/
